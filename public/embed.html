嵌入程式碼

<iframe
  src="https://your-domain.com/?mode=embed"
  allow="microphone"
  style="width: 100%; height: 600px; border: none; border-radius: 12px;"
></iframe>

<script>
const iframe = document.querySelector('iframe');

// 設定目標語言
iframe.contentWindow.postMessage({
  type: 'SET_TARGET_LANG',
  payload: { lang: 'en' }
}, '*');

// 接收翻譯結果
window.addEventListener('message', (event) => {
  if (event.data?.type === 'TRANSLATION_RESULT') {
    console.log('翻譯結果:', event.data.payload);
    // payload: { source, translation, detectedLang, targetLang }
  }
});
</script>
